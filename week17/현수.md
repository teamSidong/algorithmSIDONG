## 예산
### https://school.programmers.co.kr/learn/courses/30/lessons/12982?language=javascript
```
function solution(d, budget) {
    var answer = 0;
    d.sort(function(a, b) {
        return a - b;
    });
    d.forEach(function(val, i){
        budget = budget - val;
        if(budget < 0){
            return answer;
        }else{
            answer++;
        }
    });
    return answer;
}
```
