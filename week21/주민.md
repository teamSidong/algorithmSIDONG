        }
        if(myStr.includes(ableArr[1])){
            cnt ++;
        }
        if(myStr.includes(ableArr[2])){
            cnt ++;
        }
        if(myStr.includes(ableArr[3])){
            cnt ++;
        }
        
        if(cnt > 0){
            myStr = myStr.replace(ableArr[0],"")
            myStr = myStr.replace(ableArr[1],"")
            myStr = myStr.replace(ableArr[2],"")
            myStr = myStr.replace(ableArr[3],"")
        }
        
        console.log(myStr);
        if(myStr.length != 0){
            cnt = 0;
        }
        
        cntArr[idx] = cnt;
        
    })
    
    console.log(cntArr);
    
    return Math.max(...cntArr);
}

function solution(a, b) {
    var answer = 0;
    var maximum =0;
    
    //최대공약수
    for (let i = 1; i <= a; i++) {
        if(a%i === 0 && b%i === 0) {
            maximum = i
        }
    }
    
    //기약분수로 만든 분모
    b = b/maximum
    
    for(var i=2; i<=b; i++){
        var num = i;
        while(b%num === 0){
            if(b/num === 1) break;
            b = b/num;
        }
    }
    
    if(b == 2 || b == 5){
        return 1;
    }else{
        return 2;
    }
    
}
