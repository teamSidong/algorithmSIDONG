## 1. 문자열 여러번 뒤집기
```
function solution(my_string, queries) {
    queries.forEach( qur => {
        const a = my_string.slice(0,qur[0]);
        const b = my_string.slice(qur[0], qur[1]+1).split("").reverse().join("");
        const c = my_string.slice(qur[1]+1,my_string.length+1)
        my_string = a + b + c
    })
    return my_string;
}
```
