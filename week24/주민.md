function solution(arr1, arr2) {
    var answer = [];
    
    for(let i=0; i<arr1.length; i++){
        item1 = arr1[i];
        item2 = arr2[i];
        answer.push(new Array);
        for(let j=0; j<item1.length; j++){
            value1 = item1[j]
            value2 = item2[j]
            answer[i].push(value1 + value2);
        }
    }
    
    return answer;
}
